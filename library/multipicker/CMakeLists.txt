cmake_minimum_required(VERSION 3.22.0)
project(multipicker LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
file(GLOB_RECURSE SOURCES "*.cpp" "*.hpp")
add_library(multipicker STATIC ${SOURCES})
target_include_directories(multipicker PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(multipicker PUBLIC progressive)
if(ANDROID)
    target_link_libraries(multipicker PUBLIC android log)
endif()

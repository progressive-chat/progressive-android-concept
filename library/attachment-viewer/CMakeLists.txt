cmake_minimum_required(VERSION 3.22.0)
project(attachment_viewer LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
file(GLOB_RECURSE SOURCES "*.cpp" "*.hpp")
add_library(attachment_viewer STATIC ${SOURCES})
target_include_directories(attachment_viewer PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(attachment_viewer PUBLIC progressive)
if(ANDROID)
    target_link_libraries(attachment_viewer PUBLIC android log)
endif()

cmake_minimum_required(VERSION 3.22.0)
project(autocomplete LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
file(GLOB_RECURSE SOURCES "*.cpp" "*.hpp")
add_library(autocomplete STATIC ${SOURCES})
target_include_directories(autocomplete PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(autocomplete PUBLIC progressive)
if(ANDROID) target_link_libraries(autocomplete PUBLIC android log) endif()

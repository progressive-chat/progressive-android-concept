cmake_minimum_required(VERSION 3.22.0)
project(barcodescanner LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
file(GLOB_RECURSE SOURCES "*.cpp" "*.hpp")
add_library(barcodescanner STATIC ${SOURCES})
target_include_directories(barcodescanner PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
if(ANDROID) target_link_libraries(barcodescanner PUBLIC android log) endif()
